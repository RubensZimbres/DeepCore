dg::deepcore::Value Class Referencefinal
Processing » Type System

Detailed Description

A class whose instance can hold instances of any type.

Example

Value v1 = (int32_t)42;
assert(v.cast<int32_t>() == 42);
try {
// This will throw an exception
auto x = v.cast<int16_t>();
} catch (const Error& e) {
// .....
}
assert(v.convert<int16_t>() == 42);
v2 = v1.convert(types::Int16);
assert(v.cast<int16_t>() == 42);
auto v3 = v2.moveConvert(types::Float32);
assert(v3.cast<float>() == 42.0)
assert(v2.type() == types::Void);
